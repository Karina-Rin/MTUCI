# работаем в терминале MSYS2 UCRT64

# переходим в директорию
cd /C/Dev/Mtuci/MTUCI/RiPV

# запускаем компиляцию, создается файл с расширением .exe
g++ 1_time.cpp -o 1_time -fopenmp

# запуск
./1_time.exe

# запись в файл
./1_time.exe >> 1.txt
